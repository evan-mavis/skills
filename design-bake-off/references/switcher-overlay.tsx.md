# Switcher Overlay — Starting Code

Copy the TSX below into the switcher path chosen in Step 5 of the workflow. Rename `PageNameVariantSwitcher` to match the target page (e.g., `SellerPlacementRequestVariantSwitcher`). Update the `VARIANTS` import paths and entries after the variant subagents finish.

The switcher:

- Floats in the bottom-right corner.
- Shows the current variant name and index (`3 / 15`).
- Cycles with `←` / `→` arrow keys.
- Jumps to a variant by typing its number (`1`–`9`, then `0` for 10, `Shift+1` for 11, etc.) or by clicking from the dropdown.
- Has an "Original" entry that maps to `null` so the user can return to the unmodified page.
- Persists the selection in `localStorage` so reloads keep the chosen variant.
- Is fully self-contained — no external UI library, no design-system imports — so it never collides with whatever the variants are using.

```tsx
import { useCallback, useEffect, useMemo, useState } from 'react';
import type { ComponentType } from 'react';

import type { VariantProps } from './types';

// Subagents do not edit this file. After the bake-off, the orchestrator fills
// in the VARIANTS registry below — one entry per generated variant file.
import RequestVariant01CardStack from './RequestVariant01CardStack';
// import RequestVariant02KeyValueTable from './RequestVariant02KeyValueTable';
// ...add the rest after the bake-off completes.

export type VariantEntry = {
  id: string;
  label: string;
  Component: ComponentType<VariantProps>;
};

const VARIANTS: VariantEntry[] = [
  { id: '01-card-stack', label: '01 · Card Stack', Component: RequestVariant01CardStack },
  // { id: '02-key-value-table', label: '02 · Key/Value Table', Component: RequestVariant02KeyValueTable },
  // ...
];

const STORAGE_KEY = 'design-bake-off:selected-variant';

type Props = {
  value: string | null;
  onChange: (id: string | null) => void;
};

export function PageNameVariantSwitcher({ value, onChange }: Props) {
  const [open, setOpen] = useState(false);

  useEffect(() => {
    const stored = window.localStorage.getItem(STORAGE_KEY);
    if (stored && stored !== value && (stored === 'original' || VARIANTS.some((v) => v.id === stored))) {
      onChange(stored === 'original' ? null : stored);
    }
  }, [onChange, value]);

  useEffect(() => {
    window.localStorage.setItem(STORAGE_KEY, value ?? 'original');
  }, [value]);

  const currentIndex = useMemo(() => {
    if (value === null) return -1;
    return VARIANTS.findIndex((v) => v.id === value);
  }, [value]);

  const cycle = useCallback(
    (delta: number) => {
      const total = VARIANTS.length + 1; // +1 for "Original"
      const fromIndex = currentIndex + 1; // shift so "Original" = 0
      const nextIndex = (fromIndex + delta + total) % total;
      onChange(nextIndex === 0 ? null : VARIANTS[nextIndex - 1].id);
    },
    [currentIndex, onChange],
  );

  useEffect(() => {
    const handler = (e: KeyboardEvent) => {
      const target = e.target as HTMLElement | null;
      if (target && (target.tagName === 'INPUT' || target.tagName === 'TEXTAREA' || target.isContentEditable)) {
        return;
      }
      if (e.key === 'ArrowRight') {
        e.preventDefault();
        cycle(1);
      } else if (e.key === 'ArrowLeft') {
        e.preventDefault();
        cycle(-1);
      } else if (e.key === '0') {
        onChange(null);
      } else if (/^[1-9]$/.test(e.key)) {
        const idx = Number.parseInt(e.key, 10) - 1;
        if (VARIANTS[idx]) onChange(VARIANTS[idx].id);
      }
    };
    window.addEventListener('keydown', handler);
    return () => window.removeEventListener('keydown', handler);
  }, [cycle, onChange]);

  const currentLabel = value === null ? 'Original' : VARIANTS.find((v) => v.id === value)?.label ?? 'Unknown';
  const positionLabel = value === null ? `0 / ${VARIANTS.length}` : `${currentIndex + 1} / ${VARIANTS.length}`;

  return (
    <div
      style={{
        position: 'fixed',
        bottom: 16,
        right: 16,
        zIndex: 2147483647,
        fontFamily: 'ui-sans-serif, system-ui, -apple-system, sans-serif',
        fontSize: 12,
        color: '#0f172a',
        background: 'rgba(255, 255, 255, 0.96)',
        backdropFilter: 'blur(8px)',
        border: '1px solid rgba(15, 23, 42, 0.12)',
        borderRadius: 12,
        boxShadow: '0 10px 30px rgba(15, 23, 42, 0.18)',
        padding: 8,
        minWidth: 220,
        userSelect: 'none',
      }}
    >
      <div style={{ display: 'flex', alignItems: 'center', gap: 6 }}>
        <button
          type="button"
          onClick={() => cycle(-1)}
          aria-label="Previous variant"
          style={buttonStyle}
        >
          ←
        </button>
        <button
          type="button"
          onClick={() => setOpen((o) => !o)}
          style={{
            ...buttonStyle,
            flex: 1,
            justifyContent: 'space-between',
            paddingLeft: 10,
            paddingRight: 10,
            display: 'flex',
            alignItems: 'center',
            gap: 8,
          }}
        >
          <span style={{ fontWeight: 600 }}>{currentLabel}</span>
          <span style={{ opacity: 0.5, fontVariantNumeric: 'tabular-nums' }}>{positionLabel}</span>
        </button>
        <button
          type="button"
          onClick={() => cycle(1)}
          aria-label="Next variant"
          style={buttonStyle}
        >
          →
        </button>
      </div>
      {open && (
        <div
          role="listbox"
          style={{
            marginTop: 8,
            maxHeight: 320,
            overflowY: 'auto',
            borderTop: '1px solid rgba(15, 23, 42, 0.08)',
            paddingTop: 6,
          }}
        >
          <VariantRow
            label="Original"
            active={value === null}
            onSelect={() => {
              onChange(null);
              setOpen(false);
            }}
          />
          {VARIANTS.map((v) => (
            <VariantRow
              key={v.id}
              label={v.label}
              active={value === v.id}
              onSelect={() => {
                onChange(v.id);
                setOpen(false);
              }}
            />
          ))}
        </div>
      )}
      <div style={{ marginTop: 8, opacity: 0.55, fontSize: 10, lineHeight: 1.4 }}>
        ← / → to cycle · 1–9 to jump · 0 for original
      </div>
    </div>
  );
}

const buttonStyle: React.CSSProperties = {
  appearance: 'none',
  border: '1px solid rgba(15, 23, 42, 0.12)',
  background: 'white',
  color: '#0f172a',
  borderRadius: 8,
  padding: '6px 10px',
  cursor: 'pointer',
  fontSize: 12,
  lineHeight: 1,
};

function VariantRow({ label, active, onSelect }: { label: string; active: boolean; onSelect: () => void }) {
  return (
    <button
      type="button"
      role="option"
      aria-selected={active}
      onClick={onSelect}
      style={{
        display: 'block',
        width: '100%',
        textAlign: 'left',
        padding: '6px 10px',
        margin: 0,
        background: active ? 'rgba(15, 23, 42, 0.06)' : 'transparent',
        border: 'none',
        borderRadius: 6,
        cursor: 'pointer',
        fontSize: 12,
        color: '#0f172a',
        fontWeight: active ? 600 : 400,
      }}
    >
      {label}
    </button>
  );
}

export default PageNameVariantSwitcher;
```

## Integration snippet for the page

```tsx
import { useState } from 'react';
import { PageNameVariantSwitcher, VARIANTS } from './components/page-name-variants/PageNameVariantSwitcher';

export function PageName() {
  const [variantId, setVariantId] = useState<string | null>(null);
  const sharedProps = { /* whatever the original page passes down */ };

  const Variant = variantId
    ? VARIANTS.find((v) => v.id === variantId)?.Component
    : null;

  return (
    <>
      {Variant ? <Variant {...sharedProps} /> : <OriginalPageBody {...sharedProps} />}
      {import.meta.env.DEV && (
        <PageNameVariantSwitcher value={variantId} onChange={setVariantId} />
      )}
    </>
  );
}
```

For Next.js / CRA replace `import.meta.env.DEV` with `process.env.NODE_ENV !== 'production'`.

## Why this design

- **One-file integration.** The page needs only `useState` + the switcher + a conditional render. No context, no router, no providers.
- **Persistence across reloads.** A localStorage round-trip means the user can refresh and keep iterating on the same variant.
- **Keyboard-first cycling.** Reviewing 15 variants is faster with `←` / `→` than clicking.
- **Inline styles.** Variants will probably use Tailwind, CSS-in-JS, or a design-system library. The switcher uses raw inline styles so its appearance does not depend on any of that and cannot be unintentionally restyled.
- **Maximum z-index.** Guarantees the switcher floats above any sticky header or modal in the variants.
